1.1)    SELECT BOOKS.Id, BOOKS.Title, BOOKS.NumberOfTomes, AUTHOR.FirstName, AUTHOR.SecondName
	FROM BOOKS INNER JOIN AUTHOR ON BOOKS.AuthorID = AUTHOR.ID;

  2)    SELECT BOOKS.Id, BOOKS.Title, BOOKS.NumberOfTomes, AUTHOR.FirstName, AUTHOR.SecondName
	FROM BOOKS INNER JOIN AUTHOR ON BOOKS.AuthorID = AUTHOR.ID WHERE AUTHOR.FirstName = 'Sample' AND AUTHOR.SecondName = "Sample2";

2.1)    SELECT BOOKS.Id, BOOKS.Title, GENRE.Genre 
	FROM BOOKS INNER JOIN BOOKS_GENRE ON BOOKS.Id = BOOKS_GENRE.BookId INNER JOIN GENRES ON BOOKS_GENRE.GenreId = GENRES.Id;

  2)    SELECT BOOKS.Id, BOOKS.Title, GENRE.Genre 
	FROM BOOKS INNER JOIN BOOKS_GENRE ON BOOKS.Id = BOOKS_GENRE.BookId INNER JOIN GENRES ON BOOKS_GENRE.GenreId = GENRES.Id WHERE GENRE.Genre = 'Sample';

3.1)	SELECT GENRE.Genre, COUNT(*) 
	FROM BOOKS INNER JOIN BOOKS_GENRE ON BOOKS.Id = BOOKS_GENRE.BookId INNER JOIN GENRE ON BOOKS_GENRE.GenreId = GENRE.Id GROUP BY GENRES.Genre;

4.1)	SELECT AUTHOR.FirstName, AUTHOR.SecondName, COUNT(*) FROM BOOKS INNER JOIN AUTHOR ON BOOKS.AuthorId = AUTHOR.Id GROUP BY AUTHOR.Id;


5.1)	SELECT BOOKS.Id, BOOKS.Title, BOOKS.NumberOfTomes, GENRE.Genre FROM BOOKS INNER JOIN BOOKS_GENRE 
	ON BOOKS.Id = BOOKS_GENRE.BookId INNER JOIN GENRE ON 
	BOOKS_GENRE.GenreId = GENRE.Id WHERE BOOKS.NumberOfTomes > 5 AND BOOKS.NumberOfTomes < 11 AND GENRE.Genre = 'Fantasy' GROUP BY BOOKS.Id;
